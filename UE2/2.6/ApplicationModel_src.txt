# Anwendungsschicktprotokoll ATM->Banksystem
# ATM = Sendender Host
# Bank = Empfangender Host

#Anwendungsschicht = L5
#Transportschicht = L4
#Vermittlungsschicht = L3
#Sicherunggschicht = L2
#Netzzugangsschicht = L1

opt Second Iteration
ATM_L5->ATM_L4: Request Money\n>Encrypt Value\n>Decode Request
ATM_L4->ATM_L3: Setup TCP
ATM_L3->ATM_L2: Enable IP
ATM_L2->ATM_L1: Setup Frames
ATM_L1->Network_A: Send Bits via Medium

loop
    Network_A->Network_B: Routing
    Network_B->Network_A: Routing
end

Network_B->Bank_L1: Receiving Data via Medium
Bank_L1->Bank_L2: Setup Frames
Bank_L2->Bank_L3: Read IP-Sender
    Bank_L3-->ATM_L3: Wrong receiver
Bank_L3->Bank_L4: Read TCP-Protokoll
    Bank_L4-->ATM_L4: TCP-Paket received 
Bank_L4->Bank_L4: Check on Databanks
alt Kunde hat Geld
Bank_L4-->ATM_L5: Kunde hat Geld
else Kunde hat zuwenig Geld
Bank_L4-->ATM_L5: Kunde hat zuwenig Geld
end
